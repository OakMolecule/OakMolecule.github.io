<script>
(function () {
  function updateCursor() {
    var input = document.getElementById('not-found-cmd');
    var mirror = document.getElementById('not-found-mirror');
    var cursor = document.getElementById('not-found-cursor');
    if (!input || !mirror || !cursor) return;

    var pos = input.selectionStart == null ? input.value.length : input.selectionStart;
    var before = input.value.substring(0, pos);
    mirror.textContent = before || '\u00a0';
    cursor.style.left = mirror.offsetWidth + 'px';

    mirror.textContent = input.value || '\u00a0';
    input.style.width = Math.max(mirror.offsetWidth + 4, 56) + 'px';
  }

  function init404Terminal() {
    var input = document.getElementById('not-found-cmd');
    if (!input || input.dataset.bound) return;
    input.dataset.bound = '1';

    ['input', 'keyup', 'click', 'focus', 'select'].forEach(function (evt) {
      input.addEventListener(evt, updateCursor);
    });
    document.addEventListener('selectionchange', updateCursor);

    input.addEventListener('keydown', function (e) {
      if (e.key !== 'Enter') return;
      //if (input.value.trim() === 'cd /') {
        window.location.href = '/';
      //}
    });

    input.focus();
    input.setSelectionRange(input.value.length, input.value.length);
    updateCursor();
  }

  document.addEventListener('DOMContentLoaded', init404Terminal);
  document.addEventListener('page:loaded', init404Terminal);
})();
</script>
